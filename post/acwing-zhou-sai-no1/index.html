Acwing 3578 最大中位数
#include
#include
#include

using namespace std;

typedef long long LL；//由于数据范围所以采用LL;

const int N = 200010;

int n, m;
int w[N];

bool check(int mid)
{
LL res = 0;
for (int i = n / 2; i < n; i ++ )
if (w[i] < mid)
res += mid - w[i];
return res <= m;
}

int main()
{
scanf("%d%d", &n, &m);
for (int i = 0; i < n; i ++ ) scanf("%d", &w[i]);
sort(w, w + n);//将所有数据排序
int l = 0, r = 2e9;
//二分排序
while (l < r)
{
int mid = (LL)l + r + 1 >> 1;
if (check(mid)) l = mid;
else r = mid - 1;
}
printf("%d\n", r);
return 0;
}

优化算法，并查集：
#include
#include
#include

using namespace std;

const int N = 200010;

int n;
int p[N], s[N];

int find(int x) // 并查集
{
if (p[x] != x) p[x] = find(p[x]);
return p[x];
}

int main()
{
int T;
scanf("%d", &T);

while (T -- )
{
    scanf("%d", &n);
    for (int i = 1; i <= n; i ++ ) p[i] = i, s[i] = 1;
    for (int i = 1; i <= n; i ++ )
    {
        int j;
        scanf("%d", &j);
        if (find(i) != find(j))
        {
            s[find(j)] += s[find(i)];
            p[find(i)] = find(j);
        }
    }
    for (int i = 1; i <= n; i ++ )
        printf("%d ", s[find(i)]);
    puts("");
}

return 0;

}
