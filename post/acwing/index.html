#include<bits/stdc++.h>
using namespace std;
using LL = long long;
int main(){
cin.tie(nullptr)->sync_with_stdio(false);
int N, K;
cin >> N >> K;
vector v(N);
for(LL& x : v) cin >> x;
sort(v.begin(), v.end(), greater());
LL L = 1, R = 2E9;
while(L < R){
LL M = (L + R + 1) >> 1;
LL need = 0;
for(int i = 0; i < (N + 1) / 2; i += 1) need += max(0LL, M - v[i]);
if(need <= K) L = M;
else R = M - 1;
}
cout << L;
return 0;
}
